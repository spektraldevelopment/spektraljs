cancelEvent()

Prevents default action triggered by event.

Testing Results

Test Elements

a - href="#ceHash"

Example Code

var
    testAnchor = Spektral.getElement("ceAnchor"),
    hash, hashCheck;

Spektral.attachEventListener(testAnchor, "click", onClickEvent);

Spektral.triggerEvent(testAnchor, "click");

function onClickEvent(evt) {

    hashCheck = Spektral.createTimeOut(5, checkForHash);

    //Stops browser from navigating to
    //href which is hash tag #ceHash
    Spektral.cancelEvent(evt);
};

function checkForHash() {

    hash = Spektral.hashDetected();
};

stopPropagation()

Stops event from bubbling up into parent elements.

    testEvent = Spektral.createEvent("testEvent"), evtTarget;

Spektral.triggerEvent(testDiv, testEvent);

function onTestEvent(evt) {

    //Without cancelPropagation this function
    //is called twice due to the event
    //bubbling up from testDiv to the document
    Spektral.cancelPropagation(evt);
};

getMousePos()

Returns and object containing:
innerX, innerY - (The x and y positions inside the target element),
pageX, pageY - (The x and y positions on the entire page/document),
screenX, screenY - (The x and y positions on the screen/monitor itself),
viewportX, viewportY - (The x and y positions in the visible
display area in the browser),
offsetX (scrollLeft), offsetY(scrollTop)

Testing Results

Test Elements

div - id="gmpDiv"

Example Code

var
    testDiv = Spektral.getElement("gmpDiv"),
    mousePos, divX, divY, pageX, pageY,
    targetID, viewX, viewY, screenX, screenY;

Spektral.attachEventListener(window, "mousemove", onMouseMove);
Spektral.attachEventListener(testDiv, "mousemove", onMouseMove);

function onMouseMove(evt) {

    targetID = Spektral.getTargetID(evt);
    mousePos = Spektral.getMousePos(evt);

    if(targetID === "gmpDiv") {
        divX = mousePos.innerX;
        divY = mousePos.innerY;
    } else {
        //Viewport
        viewX = mousePos.viewportX;
        viewY = mousePos.viewportY;

        //Screen
        screenX = mousePos.screenX;
        screenY = mousePos.screenY;

        //Page
        pageX = mousePos.pageX;
        pageY = mousePos.pageY;
        console.log("pageX: " + pageX + " pageY: " + pageY);
    }
};
