Spektral.js API Documentation

Spektral.js is a collection of helpful methods and functions to help you speed up development time. It's key goal is to utilize the latest, and best practices features of JavaScript, It is free to use under the MIT license.

Example Usage

var
    image = Spektral.getElement("exampleImage"),
    toggleButton = Spektral.getElement("toggleButton"),
    removeStyleButton = Spektral.getElement("removeStyleButton"),
    addStyleButton = Spektral.getElement("addStyleButton");

Spektral.attachEventListener(toggleButton, "click", onButtonClick);
Spektral.attachEventListener(removeStyleButton, "click", onButtonClick);
Spektral.attachEventListener(addStyleButton, "click", onButtonClick);

function onButtonClick(evt) {

    button = Spektral.getTargetID(evt);

    if (button === "toggleButton") {
        Spektral.toggleVisibility(image);
    } else if (button === "removeStyleButton") {
        Spektral.clearStyle(image);
    } else if (button === "addStyleButton") {
        Spektral.setStyle(image, "padding:10px");
    }
}
